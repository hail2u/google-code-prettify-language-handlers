lang-vim.js

" Restore previous session with cofirmation
function! s:RestoreSessionWithConfirm()
  let msg = 'Do you want to restore previous session?'

  if !argc() && confirm(msg, "&Yes\n&No", 1, 'Question') == 1
    execute 'SessionRestore'
  endif
endfunction
